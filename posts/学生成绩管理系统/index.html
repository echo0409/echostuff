学生成绩管理系统（C++实现）

Posted Oct 23, 2018 2018-10-23T19:55:27+08:00 by wyk

从文件读入一个班学生三门课（语文、数学、外语）的成绩，分别统计下列内容，要求每一个功能都用函数实现：
（1）统计平均分不及格人数并打印不及格学生名单；
（2）统计成绩在全班平均分及平均分之上的学生人数并打印其学生名单；
（3）统计平均分的各分数段的学生人数及所占的百分比；
（4）按总分成绩由高到低排出成绩的名次；
（5）打印出名次表，表格内包括学生编号、各科分数、总分和平均分；
（6）任意输入一个学号，能够查找出该学生在班级中的排名及其考试分数

类的构建

class Students
{
public:
	char name[10];
	char ID[13];
	float Chinese;
	float Math;
	float English;
	float sum;
	double aver;
	void ShowGrade();
};

class CLASS
{
public:
	Students clas[NUMBER];
	double classaverage;
	void ReadStuInfoFromFile();
	void AverageFail();
	void CompareCA();
	void Sort();
	void Print();
	void Search();
};

初始化

void CLASS::ReadStuInfoFromFile()
{
	ifstream file;
	file.open("grade.txt");  
	char str[80];
	int i = 0;
	while (file.getline(str, 80))
	{
		stringstream mystream(str);
		mystream >> clas[i].name >> clas[i].ID >> clas[i].Chinese >> clas[i].Math >> clas[i].English;
		i++;
	}
	for (int i = 0; i < NUMBER; i++)
		clas[i].aver = (clas[i].Chinese + clas[i].Math + clas[i].English) / 3.0;
	for (int i = 0; i < NUMBER; i++)
		clas[i].sum = clas[i].Chinese + clas[i].Math + clas[i].English;
	double sum = 0;
	for (int i = 0; i < NUMBER; i++)
		sum = sum + clas[i].aver;
	classaverage = sum / NUMBER;
}

统计平均分不及格人数并打印不及格学生名单

void CLASS::AverageFail()
{
	cout << "*************平均分不及格的学生名单如下*************"<<endl;
	int sum=0;
	for (int i = 0; i < NUMBER; i++)
	{
		if (clas[i].aver < 60)
		{
			sum++;
			cout << " 姓名：" << clas[i].name;
			cout << "  学号：" << clas[i].ID;
			cout << " 平均分：" << clas[i].aver<< endl;
		}
	}
	cout << "****************************************************" << endl;
}

统计成绩在全班平均分之上及平均分之上的学生人数并打印其学生名单

void CLASS::CompareCA()
{
	cout << "*************低于全班平均分的学生名单如下*************" << endl;
	int number=0;
	for (int i = 0; i < NUMBER; i++)
	{
		if (clas[i].aver < classaverage)
		{
			cout << " 姓名：" << clas[i].name;
			cout << "  学号：" << clas[i].ID;
			cout << " 平均分：" << clas[i].aver << endl;
			number++;
		}
	}
	cout << "总人数：" << number<<endl;
	cout << "******************************************************" << endl;
	cout << endl;
	cout << "************不低于全班平均分的学生名单如下************" << endl;
    number = 0;
	for (int i = 0; i < NUMBER; i++)
	{
		if (clas[i].aver >=classaverage)
		{
			cout << " 姓名：" << clas[i].name;
			cout << "  学号：" << clas[i].ID;
			cout << " 平均分：" << clas[i].aver << endl;
			number++;
		}
	}
	cout << "总人数：" << number << endl;
	cout << "******************************************************" << endl;
}

按总分成绩由高到低排出成绩的名次

void CLASS::Sort()
{
	Students t;
	for (int i = NUMBER-1; i >0; i--)
	{
		for (int j = 1; j <= i; j++)
		{
			if (clas[j+1].sum > clas[j].sum)
			{
				t = clas[j];
				clas[j] = clas[j+1];
				clas[j+1]=t;
			}
		}
	}
}

打印出名次表，表格内包括学生编号、各科分数、总分和平均分

void CLASS::Print()
{
	cout << "*********************学生名单如下*********************" << endl;
	for (int i = 0; i <NUMBER;i++)
	{
		cout << " 姓名：" << clas[i].name;
		cout << "  学号：" << clas[i].ID;
		cout << "  总分：" << clas[i].sum;
		cout << " 平均分：" << clas[i].aver << endl;
	}
	cout << "******************************************************" << endl;
}

任意输入一个学号，能够查找出该学生在班级中的排名及其考试分数

void CLASS::Search()
{
	char S[13];
	cout << "******************************************************" << endl;
	cout << "请输入要查询的学号：" << endl;
	cin >> S;
	for (int i = 0; i < NUMBER; i++)
	{
		if (!strcmp(S, clas[i].ID))
		{
			cout << " 姓名：" << clas[i].name;
			cout << "  学号：" << clas[i].ID;
			cout << "  总分：" << clas[i].sum;
			cout << " 平均分：" << clas[i].aver;
			cout<<" 排名："<<i+1 << endl;
		}
	}
	cout << "******************************************************" << endl;

}
